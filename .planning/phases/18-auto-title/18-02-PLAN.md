```go
type Config struct {
    // ... existing fields preserved ...

    // Phase 18: Auto-titler. TitlerModel falls back to LLMModel when unset; if
    // both are unset, the titler is disabled (graceful no-op). Either is
    // sufficient — the Router needs SOME model to dispatch to.

    LLMTier     string  // env: LLM_TIER  (default "free")
    TitlerModel string  // env: TITLER_MODEL (fallback: LLMModel)

    // Phase 18: Provider API keys (lifted from services/orchestrator/internal/config/config.go:31-44).
    // At least one must be set when titlerModel != "" — otherwise the titler is left disabled.
    OpenRouterAPIKey    string                // env: OPENROUTER_API_KEY
    OpenAIAPIKey        string                // env: OPENAI_API_KEY
    AnthropicAPIKey     string                // env: ANTHROPIC_API_KEY
    SelfHostedEndpoints []SelfHostedEndpoint  // env: SELF_HOSTED_{N}_URL/MODEL/API_KEY (n in 1..10)
}

type SelfHostedEndpoint struct {
    URL    string
    Model  string
    APIKey string
}
```
