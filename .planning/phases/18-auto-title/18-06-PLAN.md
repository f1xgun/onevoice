       import { memo } from 'react';

       import { api } from '@/lib/api';

       interface Conversation {

           title: string;
           titleStatus?: 'auto_pending' | 'auto' | 'manual';

           conversationId: string;
           rightSlot?: React.ReactNode;
       }

       /**
        * D-11 USER OVERRIDE structural mitigation (Landmine 1):
        *   - useQuery select projection returns a primitive string. React Query's
        *     `select` runs on cache change, but downstream consumers (this hook)
        *     receive a stable string reference unless the title changes — preventing
        *     spurious re-renders triggered by lastMessageAt or other unrelated fields.
        *   - The component is wrapped in `memo` so prop changes from ChatWindow do
        *     NOT trigger re-render unless its own props change.
        *   - This component is rendered as a SIBLING of MessageList and the composer
        *     in ChatWindow (verified via repo structure at ChatWindow.tsx:92-end).
        *     Therefore composer focus and message-list scroll are unaffected by
        *     title changes.

       function useConversationTitle(conversationId: string): string {
           const { data } = useQuery<Conversation[], Error, string>({
               queryKey: ['conversations'],
               queryFn: () => api.get('/conversations').then((r) => r.data),
               select: (list) => {
                   const conv = list.find((c) => c.id === conversationId);
                   if (!conv) return '';
                   return conv.title === '' || conv.titleStatus === 'auto_pending'
                       ? 'Новый диалог'
                       : conv.title;
               },
               enabled: !!conversationId,
           });
           return data ?? '';
       }

                   {rightSlot}
               </div>
