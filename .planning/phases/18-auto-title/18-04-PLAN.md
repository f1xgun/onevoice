       ```go
       func sanitizeTitle(raw string) string {
           s := strings.TrimSpace(raw)
           s = strings.Trim(s, `"'«»“”`)
           s = strings.TrimRight(s, ".!?;:")
           s = strings.TrimSpace(s)
           if utf8.RuneCountInString(s) > titleMaxChars {
               runes := []rune(s)
               s = string(runes[:titleMaxChars])
           }
           return s
       }
       ```
