```go
func (h *AuthHandler) cookieName() string {
	if h.secureCookies {
		return "__Secure-refresh_token"
	}
	return "refresh_token"
}

func (h *AuthHandler) setRefreshTokenCookie(w http.ResponseWriter, token string) {
	http.SetCookie(w, &http.Cookie{
		Name:     h.cookieName(),
		Value:    token,
		Path:     "/api/v1/auth",
		MaxAge:   int(7 * 24 * time.Hour / time.Second), // 604800
		HttpOnly: true,
		Secure:   h.secureCookies,
		SameSite: http.SameSiteLaxMode,
	})
}

func (h *AuthHandler) clearRefreshTokenCookie(w http.ResponseWriter) {
	http.SetCookie(w, &http.Cookie{
		Name:     h.cookieName(),
		Value:    "",
		Path:     "/api/v1/auth",
		MaxAge:   -1,
		HttpOnly: true,
		Secure:   h.secureCookies,
		SameSite: http.SameSiteLaxMode,
	})
}

func (h *AuthHandler) readRefreshTokenCookie(r *http.Request) (string, error) {
	// Try secure name first, then plain name (handles upgrade path)
	for _, name := range []string{"__Secure-refresh_token", "refresh_token"} {
		c, err := r.Cookie(name)
		if err == nil && c.Value != "" {
			return c.Value, nil
		}
	}
	return "", http.ErrNoCookie
}
```

```go
func (h *AuthHandler) RefreshToken(w http.ResponseWriter, r *http.Request) {
	refreshToken, err := h.readRefreshTokenCookie(r)
	if err != nil {
		writeJSONError(w, http.StatusUnauthorized, "missing refresh token")
		return
	}

	user, accessToken, newRefreshToken, err := h.userService.RefreshToken(r.Context(), refreshToken)
	if err != nil {
		if errors.Is(err, domain.ErrInvalidToken) {
			h.clearRefreshTokenCookie(w)
			writeJSONError(w, http.StatusUnauthorized, "invalid token")
			return
		}
		slog.Error("failed to refresh token", "error", err)
		writeJSONError(w, http.StatusInternalServerError, "internal server error")
		return
	}

	h.setRefreshTokenCookie(w, newRefreshToken)
	writeJSON(w, http.StatusOK, RefreshTokenResponse{
		User:        user,
		AccessToken: accessToken,
	})
}
```

```go
func (h *AuthHandler) Logout(w http.ResponseWriter, r *http.Request) {

	if err != nil {

}
```

<task id="1-03-03">
<title>Atomic refresh token rotation with GETDEL</title>
<read_first>
- services/api/internal/service/user.go
</read_first>
<action>
In `services/api/internal/service/user.go`, in the `RefreshToken` method:

Replace the non-atomic GET + DEL sequence (around lines 175-190):
```go
userID, err := s.redis.Get(ctx, oldKey).Result()
```
...and later...
```go
s.redis.Del(ctx, oldKey)
```

With a single atomic `GetDel` call:
```go
userID, err := s.redis.GetDel(ctx, oldKey).Result()
```

Remove the separate `s.redis.Del(ctx, oldKey)` line entirely. The `GetDel` atomically returns the value and deletes the key in one Redis command, preventing the TOCTOU race where two concurrent refresh requests could both succeed with the same token.

The error handling remains the same: `redis.Nil` means the token was already consumed (replay attempt) and should return `domain.ErrInvalidToken`.
</action>

- `services/api/internal/service/user.go` does NOT contain a standalone `s.redis.Del(ctx, oldKey)` call in the RefreshToken method (the Logout method's Del is fine)
