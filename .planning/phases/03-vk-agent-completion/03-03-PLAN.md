## Overview

Add `vk__reply_comment` (creates a threaded reply to a wall comment) and `vk__delete_comment` (removes a comment from a wall post) tools. Both use the VK SDK's `WallCreateComment` and `WallDeleteComment` methods respectively.

   ```go
       resp, err := c.vk.WallCreateComment(vkapi.Params{
           "owner_id":          groupID,
           "post_id":           postID,
           "message":           text,
           "reply_to_comment":  commentID,
       })
       if err != nil {
           return 0, fmt.Errorf("vk wall.createComment: %w", err)
       }
       return resp.CommentID, nil
   }
   ```

   ```go
   func (c *Client) DeleteComment(groupID string, commentID int) error {
       _, err := c.vk.WallDeleteComment(vkapi.Params{
           "owner_id":   groupID,
           "comment_id": commentID,
       })
       if err != nil {
           return fmt.Errorf("vk wall.deleteComment: %w", err)
       }
       return nil
   }
   ```

   ```go
   func (h *Handler) replyComment(ctx context.Context, req a2a.ToolRequest) (*a2a.ToolResponse, error) {
       postIDf, _ := req.Args["post_id"].(float64)
       postID := int(postIDf)
       if postID == 0 {
           return nil, a2a.NewNonRetryableError(fmt.Errorf("vk: post_id is required and must be > 0"))
       }
       commentIDf, _ := req.Args["comment_id"].(float64)
       commentID := int(commentIDf)
       if commentID == 0 {
           return nil, a2a.NewNonRetryableError(fmt.Errorf("vk: comment_id is required and must be > 0"))
       }
       text, _ := req.Args["text"].(string)
       if text == "" {
           return nil, a2a.NewNonRetryableError(fmt.Errorf("vk: text is required for reply"))
       }

       client, groupID, err := h.getClient(ctx, req)
       if err != nil {
           return nil, err
       }
       newCommentID, err := client.ReplyComment(groupID, postID, commentID, text)
       if err != nil {
           return nil, fmt.Errorf("vk: reply comment: %w", classifyVKError(err))
       }
       return &a2a.ToolResponse{
           TaskID:  req.TaskID,
           Success: true,
           Result:  map[string]interface{}{"comment_id": float64(newCommentID)},
       }, nil
   }
   ```
</action>
