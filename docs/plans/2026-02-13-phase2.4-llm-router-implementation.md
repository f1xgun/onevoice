**Context — what already exists:**
- `pkg/llm/provider.go` — `Provider` interface
- `pkg/llm/types.go` — `ChatRequest`, `ChatResponse`, `StreamChunk`, `ModelInfo`, `Strategy`, `StrategyCost`, `StrategySpeed`, `CostBreakdown`
- `pkg/llm/config.go` — `CommissionConfig`
- `pkg/llm/registry.go` — `Registry`, `ModelProviderEntry` (fields: `Provider`, `HealthStatus`, `Enabled`, `InputCostPer1MTok`, `OutputCostPer1MTok`, `AvgLatencyMs`)
- `pkg/llm/ratelimit.go` — `RateLimiter.CheckLimit(ctx, userID, tier, tokens) (bool, error)`
- `pkg/llm/billing.go` — `BillingRepository`, `UsageLog`, `CalculateCommission(cost, mode, tier)`
- `pkg/llm/providers/openrouter.go` — `OpenRouterProvider` (NewOpenRouter)
- `pkg/llm/providers/openai.go` — `OpenAIProvider` (NewOpenAI)
- `pkg/llm/providers/anthropic.go` — `AnthropicProvider` (NewAnthropic)

**Key design rules:**
- `StrategyCost`: select provider with lowest `(InputCostPer1MTok + OutputCostPer1MTok) / 2`
- `StrategySpeed`: select provider with lowest `AvgLatencyMs`; `AvgLatencyMs == 0` means no data, ranked last
- Skip any entry where `HealthStatus != "healthy"` OR `Enabled == false`
- Rate limit check: `rateLimiter.CheckLimit(ctx, req.UserID, tier, 0)`; skip if `UserID == uuid.Nil`
- After successful `Chat`: call `registry.RecordSuccess`, then `billing.LogUsage` (fire-and-forget)
- After failed `Chat`: call `registry.RecordFailure`
- `ChatStream`: select provider same way as `Chat`, record success/failure, no billing
- `Tier string` field added to `ChatRequest`; default to `"free"` if empty

package llm

import (
	"context"
	"errors"

	"github.com/google/uuid"
)

// Sentinel errors returned by the Router.
var (
	ErrNoProvider        = errors.New("no healthy provider available for model")
	ErrRateLimitExceeded = errors.New("rate limit exceeded")
)

// RateLimitChecker abstracts rate limit checking for testability.
// *RateLimiter satisfies this interface.
type RateLimitChecker interface {
	CheckLimit(ctx context.Context, userID uuid.UUID, tier string, tokens int) (bool, error)
}

// Router routes LLM requests to the best available provider.
type Router struct {
	registry    *Registry
	rateLimiter RateLimitChecker
	billing     BillingRepository
	providers   map[string]Provider
	commission  CommissionConfig
}

// RouterOption is a functional option for Router.
type RouterOption func(*Router)

// WithRateLimiter attaches a *RateLimiter to the Router.
func WithRateLimiter(rl *RateLimiter) RouterOption {
	return func(r *Router) { r.rateLimiter = rl }
}

// WithRateLimitChecker sets a custom RateLimitChecker (primarily for testing).
func WithRateLimitChecker(rlc RateLimitChecker) RouterOption {
	return func(r *Router) { r.rateLimiter = rlc }
}

// WithBilling attaches a BillingRepository to the Router.
func WithBilling(br BillingRepository) RouterOption {
	return func(r *Router) { r.billing = br }
}

// WithProvider registers a Provider implementation in the Router.
func WithProvider(p Provider) RouterOption {
	return func(r *Router) { r.providers[p.Name()] = p }
}

// WithCommission sets the commission configuration.
func WithCommission(cfg CommissionConfig) RouterOption {
	return func(r *Router) { r.commission = cfg }
}

// NewRouter creates a Router backed by the given Registry.
func NewRouter(registry *Registry, opts ...RouterOption) *Router {
	r := &Router{
		registry:  registry,
		providers: make(map[string]Provider),
	}
	for _, o := range opts {
		o(r)
	}
	return r
}

// pickProvider returns the best candidate entry and its Provider for the model
// and strategy. Returns ErrNoProvider if no healthy+enabled+registered provider found.
func (r *Router) pickProvider(model string, strategy Strategy) (*ModelProviderEntry, Provider, error) {
	entries := r.registry.GetModelProviders(model)

	var best *ModelProviderEntry
	var bestProvider Provider

	for _, e := range entries {
		if e.HealthStatus != "healthy" || !e.Enabled {
			continue
		}
		p, ok := r.providers[e.Provider]
		if !ok {
			continue
		}

		if best == nil {
			best = e
			bestProvider = p
			continue
		}

		if strategy == StrategyCost {
			if avgCost(e) < avgCost(best) {
				best = e
				bestProvider = p
			}
		} else {
			if betterLatency(e, best) {
				best = e
				bestProvider = p
			}
		}
	}

	if best == nil {
		return nil, nil, ErrNoProvider
	}
	return best, bestProvider, nil
}

func avgCost(e *ModelProviderEntry) float64 {
	return (e.InputCostPer1MTok + e.OutputCostPer1MTok) / 2.0
}

func betterLatency(candidate, current *ModelProviderEntry) bool {
	if candidate.AvgLatencyMs == 0 {
		return false
	}
	if current.AvgLatencyMs == 0 {
		return true
	}
	return candidate.AvgLatencyMs < current.AvgLatencyMs
}

func tierFromRequest(req ChatRequest) string {
	if req.Tier == "" {
		return "free"
	}
	return req.Tier
}

// Chat dispatches a synchronous chat completion to the selected provider.
func (r *Router) Chat(ctx context.Context, req ChatRequest) (*ChatResponse, error) {
	if r.rateLimiter != nil && req.UserID != uuid.Nil {
		tier := tierFromRequest(req)
		allowed, err := r.rateLimiter.CheckLimit(ctx, req.UserID, tier, 0)
		if err != nil {
			return nil, err
		}
		if !allowed {
			return nil, ErrRateLimitExceeded
		}
	}

	entry, provider, err := r.pickProvider(req.Model, req.Strategy)
	if err != nil {
		return nil, err
	}

	resp, err := provider.Chat(ctx, req)
	if err != nil {
		r.registry.RecordFailure(entry.Provider, req.Model)
		return nil, err
	}

	r.registry.RecordSuccess(entry.Provider, req.Model, resp.Latency)
	resp.Provider = entry.Provider

	if r.billing != nil {
		r.logBilling(ctx, req, entry, resp)
	}

	return resp, nil
}

// ChatStream dispatches a streaming chat completion to the selected provider.
func (r *Router) ChatStream(ctx context.Context, req ChatRequest) (<-chan StreamChunk, error) {
	if r.rateLimiter != nil && req.UserID != uuid.Nil {
		tier := tierFromRequest(req)
		allowed, err := r.rateLimiter.CheckLimit(ctx, req.UserID, tier, 0)
		if err != nil {
			return nil, err
		}
		if !allowed {
			return nil, ErrRateLimitExceeded
		}
	}

	entry, provider, err := r.pickProvider(req.Model, req.Strategy)
	if err != nil {
		return nil, err
	}

	ch, err := provider.ChatStream(ctx, req)
	if err != nil {
		r.registry.RecordFailure(entry.Provider, req.Model)
		return nil, err
	}

	r.registry.RecordSuccess(entry.Provider, req.Model, 0)
	return ch, nil
}

// logBilling calculates cost and logs usage. Errors are silently dropped.
func (r *Router) logBilling(ctx context.Context, req ChatRequest, entry *ModelProviderEntry, resp *ChatResponse) {
	inputCostUSD := float64(resp.Usage.InputTokens) * entry.InputCostPer1MTok / 1_000_000
	outputCostUSD := float64(resp.Usage.OutputTokens) * entry.OutputCostPer1MTok / 1_000_000
	providerCost := inputCostUSD + outputCostUSD

	tier := tierFromRequest(req)
	commission := CalculateCommission(providerCost, r.commission.Mode, tier)

	_ = r.billing.LogUsage(ctx, &UsageLog{
		ID:              uuid.New(),
		UserID:          req.UserID,
		Model:           req.Model,
		Provider:        entry.Provider,
		InputTokens:     resp.Usage.InputTokens,
		OutputTokens:    resp.Usage.OutputTokens,
		ProviderCostUSD: providerCost,
		CommissionUSD:   commission,
		UserCostUSD:     providerCost + commission,
		UserTier:        tier,
	})
}

func TestRouter_Billing_LoggedAfterSuccess(t *testing.T) {
	entry := healthyEntry("gpt-4", "openai", 1.0, 3.0, 300)
	registry := newTestRegistry(entry)

	billing := &MockBillingRepository{}
	userID := uuid.New()

	r := llm.NewRouter(registry,
		llm.WithProvider(&stubProvider{
			name: "openai",
			response: &llm.ChatResponse{
				Content: "hello",
				Usage:   llm.TokenUsage{InputTokens: 500, OutputTokens: 200, TotalTokens: 700},
			},
		}),
		llm.WithBilling(billing),
		llm.WithCommission(llm.CommissionConfig{Mode: "percentage"}),
	)

	resp, err := r.Chat(context.Background(), llm.ChatRequest{
		Model:  "gpt-4",
		UserID: userID,
		Tier:   "basic",
	})
	require.NoError(t, err)
	assert.Equal(t, "hello", resp.Content)

	logs, err := billing.GetMonthlyUsage(context.Background(), userID, 2026, 2)
	require.NoError(t, err)
	require.Len(t, logs, 1)

	log := logs[0]
	assert.Equal(t, "gpt-4", log.Model)
	assert.Equal(t, "openai", log.Provider)
	assert.Equal(t, 500, log.InputTokens)
	assert.Equal(t, 200, log.OutputTokens)
	assert.Equal(t, "basic", log.UserTier)

	// ProviderCost: (500 * 1.0 / 1_000_000) + (200 * 3.0 / 1_000_000) = 0.0005 + 0.0006 = 0.0011
	assert.InDelta(t, 0.0011, log.ProviderCostUSD, 1e-9)
	// Commission: 20% of 0.0011 = 0.00022
	assert.InDelta(t, 0.00022, log.CommissionUSD, 1e-9)
	// UserCost: 0.0011 + 0.00022 = 0.00132
	assert.InDelta(t, 0.00132, log.UserCostUSD, 1e-9)
}

## Implementation decisions

1. **`RateLimitChecker` interface** extracted from `*RateLimiter` to allow test injection without Redis.
2. **`stubProvider`** defined in test file — not a shared fixture.
3. **Billing errors silently dropped** — fire-and-forget; retry logic deferred.
4. **`ChatStream` does not log billing** — streaming token counts not available at call open time.
5. **No retry logic** — actual provider error propagated; retry is a higher-level concern.
6. **Thread safety** — Router fields set once at construction, read-only during dispatch; Registry is internally synchronized.
